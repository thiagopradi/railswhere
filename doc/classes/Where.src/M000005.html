  def to_s(format=nil)
    output=""
    
    @clauses.each_index{|index|
      omit_clause = (index==0)
      output << @clauses[index].to_s(omit_clause)  # Omit the clause if index=0
    }
    case format
    when :where
      output.empty? ? "" : " WHERE #{output}"
    else
      output
    end
  end
